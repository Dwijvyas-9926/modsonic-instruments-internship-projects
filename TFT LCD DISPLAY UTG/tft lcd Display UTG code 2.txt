   AVR128DA48 @ 24MHz  |  ST7789VI 320x240  |  TTP229 8-key

   FLOW:
     Power on → EDISON II splash → fade → INSTRUMENT UI
     KEY 8 (any time) → DIAGNOSTIC MODE (HMI panel)

   ================================================================ */

/* ================================================================
   SPI + LCD PRIMITIVES  (shared by both UIs)
   ================================================================ */
static void spi_tx(uint8_t d){
    SPI0.DATA=d; while(!(SPI0.INTFLAGS&SPI_IF_bm));
}
static void lcd_cmd(uint8_t c){ CS_LO;DC_LO;spi_tx(c);CS_HI; }
static void lcd_dat(uint8_t d){ CS_LO;DC_HI;spi_tx(d);CS_HI; }
static void win(uint16_t x0,uint16_t y0,uint16_t x1,uint16_t y1){
    lcd_cmd(0x2A);lcd_dat(x0>>8);lcd_dat(x0);lcd_dat(x1>>8);lcd_dat(x1);
    lcd_cmd(0x2B);lcd_dat(y0>>8);lcd_dat(y0);lcd_dat(y1>>8);lcd_dat(y1);
    lcd_cmd(0x2C);
}
static void fill(uint16_t x,uint16_t y,uint16_t w,uint16_t h,uint16_t c){
    if(!w||!h) return;
    win(x,y,x+w-1,y+h-1); CS_LO; DC_HI;
    for(uint32_t i=0;i<(uint32_t)w*h;i++){spi_tx(c>>8);spi_tx(c);}
    CS_HI;
}
static void hline(uint16_t x,uint16_t y,uint16_t w,uint16_t c){fill(x,y,w,1,c);}
static void vline(uint16_t x,uint16_t y,uint16_t h,uint16_t c){fill(x,y,1,h,c);}

static void apply_bright(uint8_t lv){
    lcd_cmd(0x53);lcd_dat(0x2C);
    lcd_cmd(0x51);lcd_dat((uint8_t)((uint16_t)lv*255/15));
}
static void delay_ms(uint16_t ms){
    while(ms>=100){_delay_ms(100);ms-=100;}
    while(ms>=10) {_delay_ms(10); ms-=10; }
    while(ms>=1)  {_delay_ms(1);  ms-=1;  }
}

/* ================================================================
   TTP229
   ================================================================ */
static uint8_t ttp_read(void){
    uint8_t k=0;
    for(uint8_t i=0;i<8;i++){
        TTP_SCL_LO;_delay_us(3);
        if(!TTP_SDO) k|=(1<<i);
        TTP_SCL_HI;_delay_us(3);
    }
    for(uint8_t i=0;i<8;i++){TTP_SCL_LO;_delay_us(3);TTP_SCL_HI;_delay_us(3);}
    return k;
}

static void glyph(uint16_t x,uint16_t y,char c,
                  uint16_t fg,uint16_t bg,uint8_t sc){
    if(c<32||c>126){fill(x,y,5*sc,7*sc,bg);return;}
    for(uint8_t r=0;r<7;r++){
        uint8_t b=pgm_read_byte(&FONT[(uint8_t)(c-32)][r]);
        for(uint8_t col=0;col<5;col++)
            fill(x+col*sc,y+r*sc,sc,sc,(b&(0x10>>col))?fg:bg);
    }
}
static void dtext(uint16_t x,uint16_t y,const char*s,
                  uint16_t fg,uint16_t bg,uint8_t sc){
    while(*s){glyph(x,y,*s,fg,bg,sc);x+=(uint16_t)(5+1)*sc;s++;}
}
static void numw(uint16_t x,uint16_t y,uint16_t n,uint8_t d,
                 uint16_t fg,uint16_t bg,uint8_t sc){
    char b[6]; uint8_t i=d; b[i]=0;
    while(i--){b[i]=(char)('0'+n%10);n/=10;}
    dtext(x,y,b,fg,bg,sc);
}
/* Bold: draw glyph then overdraw 1px right — thickens strokes */
static void glyph_bold(uint16_t x,uint16_t y,char c,
                       uint16_t fg,uint16_t bg,uint8_t sc){
    if(c<32||c>126){fill(x,y,(5+1)*sc,7*sc,bg);return;}
    /* background first */
    fill(x,y,(5+1)*sc,7*sc,bg);
    for(uint8_t r=0;r<7;r++){
        uint8_t b=pgm_read_byte(&FONT[(uint8_t)(c-32)][r]);
        for(uint8_t col=0;col<5;col++){
            if(b&(0x10>>col)){
                fill(x+col*sc,    y+r*sc,sc,sc,fg);
                fill(x+col*sc+1,  y+r*sc,sc,sc,fg); /* +1px bold stroke */
            }
        }
    }
}
static void dtext_bold(uint16_t x,uint16_t y,const char*s,
                       uint16_t fg,uint16_t bg,uint8_t sc){
    while(*s){glyph_bold(x,y,*s,fg,bg,sc);x+=(uint16_t)(5+2)*sc;s++;}
}
static void numw_bold(uint16_t x,uint16_t y,uint16_t n,uint8_t d,
                      uint16_t fg,uint16_t bg,uint8_t sc){
    char b[6]; uint8_t i=d; b[i]=0;
    while(i--){b[i]=(char)('0'+n%10);n/=10;}
    dtext_bold(x,y,b,fg,bg,sc);
}

    }
}

    win(x,y,x+DIG_W-1,y+DIG_H-1); CS_LO; DC_HI;
    for(uint8_t row=0;row<DIG_H;row++)
        for(uint8_t col=0;col<DIG_W;col++){

            spi_tx(c>>8);spi_tx(c);
        }
    CS_HI;
}

/* Battery % small font — no longer needed in old position, kept for smfont */
static void draw_batt_pct(uint8_t pct,uint16_t fg){
    uint8_t ch[4]; uint8_t nc=0;
    if(pct>=100){ch[nc++]=1;ch[nc++]=0;ch[nc++]=0;}
    else if(pct>=10){ch[nc++]=pct/10;ch[nc++]=pct%10;}
    else ch[nc++]=pct;
    ch[nc++]=10;
    win(PCT_X,PCT_Y,PCT_X+PCT_W-1,PCT_Y+SF_H-1); CS_LO; DC_HI;
    for(uint8_t row=0;row<SF_H;row++){
        uint8_t cp=0;
        for(uint8_t ci=0;ci<nc;ci++){
            uint16_t base=(uint16_t)ch[ci]*SF_BPD;
            for(uint8_t col=0;col<SF_W;col++){
                uint16_t bi=base+(uint16_t)row*SF_BPR+(col>>3);
                uint8_t  bv=pgm_read_byte(&smfont[bi]);
                uint16_t c=(bv&(0x80>>(col&7)))?fg:WHITE;
                spi_tx(c>>8);spi_tx(c); cp++;
            }
            if(ci<nc-1){spi_tx(WHITE>>8);spi_tx(WHITE);spi_tx(WHITE>>8);spi_tx(WHITE);cp+=2;}
        }
        while(cp<PCT_W){spi_tx(WHITE>>8);spi_tx(WHITE);cp++;}
    }
    CS_HI;
}

/* Splash: draw static EDISON II, fade out, draw UI, fade in */
static void run_splash(void){
    /* Draw splash while dark */
    win(0,0,SCR_W-1,SCR_H-1); CS_LO; DC_HI;
    for(uint16_t row=0;row<SCR_H;row++){
        uint16_t rb=row*SP_BPR;
        for(uint8_t b2=0;b2<SP_BPR;b2++)
            sp_row[b2]=pgm_read_byte(&sp_bmp[rb+b2]);
        for(uint8_t b2=0;b2<SP_BPR;b2++){
            uint8_t v=sp_row[b2];
            spi_tx((v&0x80)?0x00:0xFF);spi_tx((v&0x80)?0x00:0xFF);
            spi_tx((v&0x40)?0x00:0xFF);spi_tx((v&0x40)?0x00:0xFF);
            spi_tx((v&0x20)?0x00:0xFF);spi_tx((v&0x20)?0x00:0xFF);
            spi_tx((v&0x10)?0x00:0xFF);spi_tx((v&0x10)?0x00:0xFF);
            spi_tx((v&0x08)?0x00:0xFF);spi_tx((v&0x08)?0x00:0xFF);
            spi_tx((v&0x04)?0x00:0xFF);spi_tx((v&0x04)?0x00:0xFF);
            spi_tx((v&0x02)?0x00:0xFF);spi_tx((v&0x02)?0x00:0xFF);
            spi_tx((v&0x01)?0x00:0xFF);spi_tx((v&0x01)?0x00:0xFF);
        }
    }
    CS_HI;

    /* Hold 2 seconds */
    for(uint8_t i=0;i<20;i++) _delay_ms(100);

    /* Enable brightness control, fade out splash */
    lcd_cmd(0x53);lcd_dat(0x2C);
    lcd_cmd(0x51);lcd_dat(0xFF);
    for(int16_t b=255;b>=0;b-=8){lcd_cmd(0x51);lcd_dat((uint8_t)b);delay_ms(18);}
    lcd_cmd(0x51);lcd_dat(0x00);

    /* Draw instrument UI while dark */
    draw_instrument_ui();

    /* Fade in */
    for(uint16_t b=0;b<=255;b+=8){lcd_cmd(0x51);lcd_dat((uint8_t)b);delay_ms(18);}
    lcd_cmd(0x51);lcd_dat(0xFF);

    /* Disable brightness control — restores normal colour pipeline */
    lcd_cmd(0x53);lcd_dat(0x00);
}

/* ================================================================
   MAIN
   ================================================================ */
int main(void){
    clk24();       /* clock MUST be set before everything else */
    hw_init();
    lcd_init();

    /* Splash → instrument UI */
    run_splash();

    /* Instrument mode init */
    inst_count=0; inst_batt=100;
    uint8_t bt=0;
    prev_keys=ttp_read();

    while(1){
        uptime++;
        uint8_t keys=ttp_read();
        uint8_t np  =keys&~prev_keys;
        prev_keys=keys;

        /* Hold + press tracking (shared) */
        for(uint8_t k=0;k<8;k++){
            if(keys&(1<<k)){if(key_hold[k]<255)key_hold[k]++;}
            else key_hold[k]=0;
            if(np&(1<<k)) key_cnt[k]++;
        }

        /* ── K8 ANYWHERE: toggle between instrument and diagnostic ── */
        if(np&0x80){
            if(mode==0){
                /* Instrument → Diagnostic */
                mode=1;
                diag_splash();
                uint8_t pk2=0;
                while(1){
                    uint8_t k2=ttp_read();
                    if((k2&0x10)&&!(pk2&0x10)) break;
                    pk2=k2; _delay_ms(50);
                }
                diag_go(SCR_DMENU);
                prev_keys=ttp_read();
            } else {
                /* Diagnostic → Instrument (from anywhere, any page) */
                if(diag_page==SCR_BRIGHT){lcd_cmd(0x53);lcd_dat(0x00);}
                mode=0;
                draw_instrument_ui();
                p0=p1=p2=p3=pc=0xFF;
                bt=0; prev_keys=ttp_read();
            }
            continue;
        }

        /* ── INSTRUMENT MODE ──────────────────────── */
        if(mode==0){
            /* Auto-count: advances every 500ms = every 10 x 50ms ticks */
            static uint8_t count_tick=0;
            if(++count_tick>=10){
                count_tick=0;
                inst_count++;
                if(inst_count>9999) inst_count=0;
            }

            /* Battery: degrades every 500ms = every 10 x 50ms ticks */
            if(++bt>=10){
                bt=0;
                if(inst_batt>0) inst_batt--; else inst_batt=100;
                draw_battery(inst_batt);
            }

            /* Draw only changed digits */
            uint8_t d0=(inst_count/1000)%10,d1=(inst_count/100)%10;
            uint8_t d2=(inst_count/10)%10,  d3= inst_count%10;
            uint8_t sc=(inst_count>=100)?1:0;
            if(d0!=p0){draw_digit(D0_X,DIG_Y,d0,BLACK,WHITE);p0=d0;}
            if(d1!=p1){draw_digit(D1_X,DIG_Y,d1,BLACK,WHITE);p1=d1;}
            if(d2!=p2){draw_digit(D2_X,DIG_Y,d2,BLACK,WHITE);p2=d2;}
            if(d3!=p3){draw_digit(D3_X,DIG_Y,d3,BLACK,WHITE);p3=d3;}
            if(sc!=pc){draw_colon(sc?I_DARK:WHITE);pc=sc;}

            _delay_ms(50);   /* short poll — K8 responds within 50ms */
        }

        /* ── DIAGNOSTIC MODE ─────────────────────── */
        else {
            /* K6: back one level, or back to instrument from menu */
            if(np&0x20){
                if(diag_page==SCR_DMENU){
                    if(diag_page==SCR_BRIGHT){lcd_cmd(0x53);lcd_dat(0x00);}
                    mode=0;
                    draw_instrument_ui();
                    p0=p1=p2=p3=pc=0xFF;
                    bt=0; prev_keys=ttp_read();
                } else {
                    diag_go(SCR_DMENU);
                }
                _delay_ms(20); continue;
            }

            switch(diag_page){
                case SCR_DMENU: {
                    uint8_t oc=menu_cur;
                    if(np&0x01){if(menu_cur>0)menu_cur--;}
                    if(np&0x02){if(menu_cur<4)menu_cur++;}
                    if(np&(0x01|0x02)) menu_move(oc);
                    if(np&0x10){
                        if(MPAGE[menu_cur]==0xFF){
                            /* "Exit to Instrument" menu item */
                            if(diag_page==SCR_BRIGHT){lcd_cmd(0x53);lcd_dat(0x00);}
                            mode=0;
                            draw_instrument_ui();
                            p0=p1=p2=p3=pc=0xFF;
                            bt=0; prev_keys=ttp_read();
                            continue;
                        }
                        diag_go(MPAGE[menu_cur]);
                    }
                    break;
                }
                case SCR_BRIGHT: {
                    static uint8_t bth=0; bth++;
                    uint8_t ch=0;
                    if((np&0x01)||(bth%5==0&&(keys&0x01))){if(bright_val<15){bright_val++;ch=1;}}
                    if((np&0x02)||(bth%5==0&&(keys&0x02))){if(bright_val>0) {bright_val--;ch=1;}}
                    if(ch){apply_bright(bright_val);draw_bright_ui();}
                    break;
                }
                case SCR_BTNTEST:
                    btntest_update(keys);
                    break;
                case SCR_VOLUME: {
                    uint8_t ch=0;
                    if(keys&0x08){if(vol_val<100){vol_val++;ch=1;}}
                    if(keys&0x04){if(vol_val>0)  {vol_val--;ch=1;}}
                    if(ch) draw_vol_ui();
                    break;
                }
                case SCR_ABOUT:
                    break;
            }

            if(uptime%33==0) uptime_tick();
            _delay_ms(30);
        }
    }
