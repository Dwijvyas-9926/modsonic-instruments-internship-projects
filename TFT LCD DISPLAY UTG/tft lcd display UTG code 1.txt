/* ================================================================
   MODSONIC EDISON-II  —  INTEGRATED FIRMWARE  v3.0
   AVR128DA48 @ 24MHz

   HARDWARE (new PCB):
     TFT : ST7789VI 320x240  via SPI
           PD0=DC  PD1=RST  PD2=CS  PA4=MOSI  PA6=SCK
     KEYS: TTP229 8-key capacitive
           PB1=SCL  PB0=SDO
     POWER: PA0=power button (ISR, falling edge)
            PA5=power latch (HIGH=on)
            PA6=low battery detect  ← NOTE: PA6 also used as SCK
                (L_BATT sampled BEFORE SPI enabled; see hw_init note)
     ULTRASONIC COUNTER BUS (new PCB routing — see NOTE below):
           PB2=Q1  PB3=Q2  PB4=Q3  PB5=Q4  PC4=Q5  PC5=Q6
           PC6=Q7  PC7=Q8(CLK10) — actual counter on PA1-PA3,PE0-PE2
           *** PCB MUST route counter lines to non-conflicting pins ***
           *** Kept as #defines — adjust to match your new PCB     ***
     MCLOCK: PD3   COU_R: PD4   ECHO: PF0   CLK_10: PD5

   KEY MAPPING (TTP229 → functions):
     K1=UP (velocity/cal up)    K2=DOWN (velocity/cal down)
     K3=MM/INCH toggle          K4=reserved
     K5=PROBE (zero cal)        K6=CAL (thickness cal)
     K7=reserved                K8=DIAGNOSTIC MODE toggle

   FLOW:
     Power on → INIT/INIT_M → TFT splash → THNESS loop
     K8 → Diagnostic HMI (brightness/button test/volume/info)
     K8 again → back to measurement

   NOTE ON COUNTER BUS PINS:
     Original: Q1-Q6=PB0-5, Q7-Q9=PD0-2, Q10-Q12=PC4-6
     New PCB:  PB0/PB1 used by TTP229, PD0-2 used by TFT.
     Counter lines REMAPPED to: Q1-Q6=PB2-5+PC4-5, Q7-Q9=PA1-3,
     Q10-Q12=PC6-7+PF1. Adjust READ_CNT() if your PCB differs.
   ================================================================ */

/* ================================================================
   TIMER FUNCTIONS  (identical to old code)
   ================================================================ */
static inline void Timer_Start(uint16_t value){
    TCB0.CNT=0; TCB0.CCMP=value;
    TCB0.INTFLAGS=TCB_CAPT_bm; timerOverflow=0;
    TCB0.CTRLA|=TCB_ENABLE_bm;
}
static inline void Timer_Stop(void){ TCB0.CTRLA&=~TCB_ENABLE_bm; }
static inline uint8_t Timer_Overflow(void){
    return (TCB0.INTFLAGS&TCB_CAPT_bm)?1:0;
}
static inline void Timer_Reload(uint16_t value){
    TCB0.CNT=0; TCB0.CCMP=value; TCB0.INTFLAGS=TCB_CAPT_bm;
}
void TIMER_OVERFLOW(void){
    TCB0.INTFLAGS=TCB_CAPT_bm;
    TCB0.CTRLA&=~TCB_ENABLE_bm;
    timerOverflow=0;
}

/* ================================================================
   EEPROM  (identical to old code)
   ================================================================ */
void EEPROM_SaveCalibration(void){
    uint8_t sum=0;
    eeprom_update_word(&ee_velo,  *uiVELO_12);
    eeprom_update_dword(&ee_delta,*ulDELTA);
    eeprom_update_word(&ee_lcdlm, *uiLCD_LM);
    eeprom_update_byte(&ee_mmif,  *MM_IF);
    eeprom_update_byte(&ee_once8, *once_8);
    sum+=(uint8_t)(*uiVELO_12&0xFF);
    sum+=(uint8_t)(*uiVELO_12>>8);
    sum+=(uint8_t)(*ulDELTA&0xFF);
    sum+=(uint8_t)((*ulDELTA>>8)&0xFF);
    sum+=(uint8_t)((*ulDELTA>>16)&0xFF);
    sum+=(uint8_t)((*ulDELTA>>24)&0xFF);
    sum+=(uint8_t)(*uiLCD_LM&0xFF);
    sum+=(uint8_t)(*uiLCD_LM>>8);
    sum+=*MM_IF; sum+=*once_8;
    eeprom_update_byte(&ee_checksum,  sum);
    eeprom_update_byte(&ee_checksum_c,sum-1);
}
uint8_t EEPROM_ValidateCalibration(void){
    uint8_t sum=0;
    uint16_t velo  =eeprom_read_word(&ee_velo);
    uint32_t delta =eeprom_read_dword(&ee_delta);
    uint16_t lcdlm =eeprom_read_word(&ee_lcdlm);
    uint8_t  mmif  =eeprom_read_byte(&ee_mmif);
    uint8_t  once8 =eeprom_read_byte(&ee_once8);
    sum+=(uint8_t)(velo&0xFF); sum+=(uint8_t)(velo>>8);
    sum+=(uint8_t)(delta&0xFF);sum+=(uint8_t)((delta>>8)&0xFF);
    sum+=(uint8_t)((delta>>16)&0xFF);sum+=(uint8_t)((delta>>24)&0xFF);
    sum+=(uint8_t)(lcdlm&0xFF);sum+=(uint8_t)(lcdlm>>8);
    sum+=mmif; sum+=once8;
    if(sum==eeprom_read_byte(&ee_checksum))
        if((sum-1)==eeprom_read_byte(&ee_checksum_c))
            return 0x00;
    return 0xFF;
}
void EEPROM_LoadCalibration(void){
    *uiVELO_12=eeprom_read_word(&ee_velo);
    *ulDELTA  =eeprom_read_dword(&ee_delta);
    *uiLCD_LM =eeprom_read_word(&ee_lcdlm);
    *MM_IF    =eeprom_read_byte(&ee_mmif);
    *once_8   =eeprom_read_byte(&ee_once8);
}

/* ================================================================
   CHECKSUM  (identical to old code)
   ================================================================ */
uint8_t SUM_A(void){
    uint8_t s=0;
    s+=*((uint8_t*)uiVELO_12);  s+=*((uint8_t*)uiVELO_12+1);
    s+=*((uint8_t*)ulDELTA);    s+=*((uint8_t*)ulDELTA+1);
    s+=*((uint8_t*)ulDELTA+2);
    s+=*((uint8_t*)uiLCD_LM);   s+=*((uint8_t*)uiLCD_LM+1);
    return s;
}
uint8_t SUM_CK(void){
    uint8_t t=SUM_A();
    if(t==*ucSUM && (t-1)==*ucSUM_C) return 0x00;
    if(EEPROM_ValidateCalibration()==0x00){EEPROM_LoadCalibration();return 0x00;}
    return 0xFF;
}
void SUM_S(void){
    EEPROM_SaveCalibration();
    uint8_t t=SUM_A(); *ucSUM=t; *ucSUM_C=t-1;
}

static void spi_tx(uint8_t d){
    SPI0.DATA=d; while(!(SPI0.INTFLAGS&SPI_IF_bm));
}
static void lcd_cmd(uint8_t c){ CS_LO;DC_LO;spi_tx(c);CS_HI; }
static void lcd_dat(uint8_t d){ CS_LO;DC_HI;spi_tx(d);CS_HI; }
static void win(uint16_t x0,uint16_t y0,uint16_t x1,uint16_t y1){
    lcd_cmd(0x2A);lcd_dat(x0>>8);lcd_dat(x0);lcd_dat(x1>>8);lcd_dat(x1);
    lcd_cmd(0x2B);lcd_dat(y0>>8);lcd_dat(y0);lcd_dat(y1>>8);lcd_dat(y1);
    lcd_cmd(0x2C);
}
static void fill(uint16_t x,uint16_t y,uint16_t w,uint16_t h,uint16_t c){
    if(!w||!h) return;
    win(x,y,x+w-1,y+h-1); CS_LO; DC_HI;
    for(uint32_t i=0;i<(uint32_t)w*h;i++){spi_tx(c>>8);spi_tx(c);}
    CS_HI;
}
static void hline(uint16_t x,uint16_t y,uint16_t w,uint16_t c){fill(x,y,w,1,c);}

/* ================================================================
   DELAY HELPERS
   ================================================================ */
static void delay_ms(uint16_t ms){
    while(ms>=100){_delay_ms(100);ms-=100;}
    while(ms>=10) {_delay_ms(10); ms-=10; }
    while(ms>=1)  {_delay_ms(1);  ms-=1;  }
}
void DELAY(uint16_t t){ volatile uint16_t i; for(i=0;i<t;i++) _delay_us(50); }
void DELAYL(uint16_t v){ uint16_t i; for(i=0;i<v;i++) DELAY(0x15); }
void DELAYS(void){
    uint16_t i,j;
    for(i=0;i<0x80;i++) for(j=0;j<0x80;j++) _delay_us(10);
}
void DELAYD(uint16_t v){ uint16_t i; for(i=0;i<v;i++) DELAY(0x15); }
void DLY_LOG(uint8_t d){ uint16_t i; for(i=0;i<d;i++) DELAY(d); }

/* ================================================================
   TTP229 KEYPAD
   ================================================================ */
static uint8_t ttp_read(void){
    uint8_t k=0;
    for(uint8_t i=0;i<8;i++){
        TTP_SCL_LO;_delay_us(3);
        if(!TTP_SDO) k|=(1<<i);
        TTP_SCL_HI;_delay_us(3);
    }
    for(uint8_t i=0;i<8;i++){TTP_SCL_LO;_delay_us(3);TTP_SCL_HI;_delay_us(3);}
    return k;
}

static void glyph(uint16_t x,uint16_t y,char c,uint16_t fg,uint16_t bg,uint8_t sc){
    if(c<32||c>126){fill(x,y,5*sc,7*sc,bg);return;}
    for(uint8_t r=0;r<7;r++){
        uint8_t b=pgm_read_byte(&FONT[(uint8_t)(c-32)][r]);
        for(uint8_t col=0;col<5;col++)
            fill(x+col*sc,y+r*sc,sc,sc,(b&(0x10>>col))?fg:bg);
    }
}
static void dtext(uint16_t x,uint16_t y,const char*s,uint16_t fg,uint16_t bg,uint8_t sc){
    while(*s){glyph(x,y,*s,fg,bg,sc);x+=(uint16_t)(5+1)*sc;s++;}
}
static void numw(uint16_t x,uint16_t y,uint16_t n,uint8_t d,uint16_t fg,uint16_t bg,uint8_t sc){
    char b[6]; uint8_t i=d; b[i]=0;
    while(i--){b[i]=(char)('0'+n%10);n/=10;}
    dtext(x,y,b,fg,bg,sc);
}
static void glyph_bold(uint16_t x,uint16_t y,char c,uint16_t fg,uint16_t bg,uint8_t sc){
    if(c<32||c>126){fill(x,y,(5+1)*sc,7*sc,bg);return;}
    fill(x,y,(5+1)*sc,7*sc,bg);
    for(uint8_t r=0;r<7;r++){
        uint8_t b=pgm_read_byte(&FONT[(uint8_t)(c-32)][r]);
        for(uint8_t col=0;col<5;col++)
            if(b&(0x10>>col)){
                fill(x+col*sc,y+r*sc,sc,sc,fg);
                fill(x+col*sc+1,y+r*sc,sc,sc,fg);
            }
    }
}
static void dtext_bold(uint16_t x,uint16_t y,const char*s,uint16_t fg,uint16_t bg,uint8_t sc){
    while(*s){glyph_bold(x,y,*s,fg,bg,sc);x+=(uint16_t)(5+2)*sc;s++;}
}

uint8_t VPATH_VLAUE(uint8_t i){
    if(i<=2)  return 0x38; if(i<=4)  return 0x37; if(i==5)  return 0x36;
    if(i<=7)  return 0x35; if(i==8)  return 0x34; if(i<=10) return 0x33;
    if(i==11) return 0x32; if(i<=13) return 0x31; if(i<=15) return 0x30;
    if(i<=17) return 0x2F; if(i<=19) return 0x2E; if(i<=21) return 0x2D;
    if(i<=23) return 0x2B; if(i==24) return 0x2A; if(i==25) return 0x29;
    if(i<=27) return 0x28; if(i<=29) return 0x27; if(i<=31) return 0x26;
    if(i<=33) return 0x25; if(i<=35) return 0x24; if(i<=37) return 0x23;
    if(i<=39) return 0x22; if(i<=41) return 0x21; if(i<=43) return 0x20;
    if(i<=45) return 0x1F; if(i<=48) return 0x1E; if(i<=51) return 0x1D;
    if(i<=53) return 0x1C; if(i<=55) return 0x1B; if(i<=59) return 0x1A;
    if(i<=62) return 0x19; if(i<=65) return 0x18; if(i<=69) return 0x17;
    if(i<=72) return 0x16; if(i<=75) return 0x15; if(i<=80) return 0x14;
    if(i<=85) return 0x13; if(i<=91) return 0x12; if(i<=95) return 0x11;
    if(i<=101)return 0x10; if(i<=106)return 0x0F; if(i<=112)return 0x0E;

    if(i<=193)return 0x04; if(i<=202)return 0x03; if(i<=212)return 0x02;
    if(i<=223)return 0x01; return 0x00;

uint8_t VNOTCH(uint16_t p){
    uint16_t d;
    if(p>0x09C4) return 0;
    if(p>0x03E8){d=DIV32(p,0x64);return 0x19-d;}
    d=DIV32(p,0x0A);
    return VPATH_VLAUE((uint8_t)d);
}
uint16_t MM(uint16_t p){ return DIV32(p,0x0A); }
uint16_t INCH(uint16_t p){ return DIV32(MUL32(p,0x0A),0x00FE); }

/* ================================================================
   ULTRASONIC MEASUREMENT  (identical logic, remapped pins — see NOTE)
   ================================================================ */
void READ_CNT(void){
    /* NOTE: On new PCB, counter lines must be routed away from
       PD0-2 (TFT) and PB0-1 (TTP229).
       ASSUMED new routing:
         Q1-Q6:  PB2-5, PC4-5
         Q7-Q9:  PA1-3  (old push button pins, now free)
         Q10-Q12:PC6-7, PF1
       Adjust these masks to match your actual PCB routing.       */
    uint8_t pA=PORTA.IN, pB=PORTB.IN, pC=PORTC.IN;
    uint16_t cv=0;
    cv  = (pB>>2)&0x0F;                    /* PB2-5 = Q1-Q4, bits 0-3  */
    cv |= (uint16_t)((pC>>4)&0x03)<<4;     /* PC4-5 = Q5-Q6, bits 4-5  */
    cv |= (uint16_t)((pA>>1)&0x07)<<6;     /* PA1-3 = Q7-Q9, bits 6-8  */
    cv |= (uint16_t)((pC>>6)&0x03)<<9;     /* PC6-7 = Q10-Q11, bits 9-10 */
    cv |= (uint16_t)(PORTF.IN&PIN1_bm)<<10;/* PF1   = Q12, bit 11      */
    ulReadCnt+=cv;
}

void CYCLE(void){
    uint16_t i;
    volatile uint8_t t=0;
    ulReadCnt=0; ECHO_F=0x01;
    for(i=0;i<100;i++){
        COU_R_1; _delay_us(50);
        COU_R_0; _delay_us(5);
        MCLOCK_1; _delay_us(2);
        MCLOCK_0;
        _delay_us(200);
        READ_CNT();
        t=PORTB.IN;
        if(PORTB.IN!=t){ECHO_F=0x00;DELAYS();return;}
        _delay_us(700);
    }
    Timer_Reload(TIMER_VAL_LONG);
}

void SUBDELTA(void){ ulCY=ulReadCnt-*ulDELTA; }

void DIR_CAL(void){
    uint16_t uiTemp,uiDiv32;
    uint32_t ulMul32;
    UP_DATE=0x00;
    ulMul32=MUL32(ulCY,*uiVELO_12);
    uiDiv32=DIV32(ulMul32,0x9C40);
    uiTemp=uiDiv32-VNOTCH(uiDiv32);
    if(*MM_IF==0x00){*uiLCD_LM=MM(uiTemp);DP_2=0x00;DP_1=0x01;}
    else             {*uiLCD_LM=INCH(uiTemp);DP_1=0x00;DP_2=0x01;}
    if(*once_8==0)   *uiLCD_LM=*uiLCD_LM-0x0B33;
    TEMP=(uint8_t)*uiLCD_LM;
}

void CMP_CAL(void){
    uint32_t ulM; uint16_t uiD,uiT;
    UP_DATE=0x00;
    ulM=MUL32(ulCY,*uiVELO_12);
    uiD=DIV32(ulM,0x9C40);
    uiT=uiD-VNOTCH(uiD);
    if(*MM_IF==0x00)*uiLCD_LM=MM(uiT);
    else            *uiLCD_LM=INCH(uiT);
}

/* ================================================================

   ================================================================ */
ISR(PORTA_PORT_vect){
    if(PORTA.INTFLAGS&INT_PIN){
        PORTA.INTFLAGS=INT_PIN;
        if(!(PORTA.IN&INT_PIN)){
            CLEAR_LCD_ISR();
            SUM_S();
            P_10_0;
            while(1);
        }
    }
}

/* ================================================================
   LOW BATTERY  (adapted from old code — uses disp_batt for TFT)
   ================================================================ */
void LO_BATT(void){
    ucBATT--;
    if(ucBATT!=0) return;
    if(UP_DATE==0x01){
        ucAUTO++;
        if(ucAUTO==0x9A){
            SUM_S(); P_10_0;
            while(1);
        }
    } else {
        ucAUTO=0x00;
    }
    ucBATT=0x02;
    /* Update displayed battery level from PA6 */
    if(READ_L_BATT){
        if(disp_batt>0) disp_batt--;
        draw_battery(disp_batt);
    }
}

/* VE_DNC / VE_UPC: use TTP keys instead of push buttons */
void VE_DNC(void){
    uint8_t ucCount=0x65;
    uint8_t k;
    while(1){
        F_INC();
        if(CK_C()==TEMP) if(*uiVELO_12!=0x270F) continue;
        DIR_C();
        if(ucCount!=0x01) ucCount-=0x02;
        DLY_LOG(ucCount);
        k=ttp_read();
        if(!(k&K_UP)) break;
    }
}
void VE_UPC(void){
    uint8_t ucCount=0x65;
    uint8_t k;
    while(1){
        F_DEC();
        if(CK_C()==TEMP) if(*uiVELO_12!=0x03E8) continue;
        DIR_C();
        if(ucCount!=0x01) ucCount-=0x02;
        DLY_LOG(ucCount);
        k=ttp_read();
        if(!(k&K_DN)) break;
    }
}

void DIS_T(void){
    uint16_t i; uint8_t k,pk=0;
    while(1){
        for(i=0;i<0xA0;i++){
            k=ttp_read(); uint8_t np=k&~pk; pk=k;
            if(np&K_UP){VE_DNC();break;}
            if(np&K_DN){VE_UPC();break;}
            if(np&K_CAL){
                DELAYL(0xFF); DELAYL(0xFF);
                k=ttp_read();
                if(k&K_CAL){
                    VELO();
                    DP_1=0x01; ECHO_F=0x00; DIR_C();
                    return;
                }
            }
            DELAYD(0x20);
        }
        if(i==0xA0) break;
    }
    ECHO_F=0x00; DIR_C();
}

void DIST_C(void){
    ECHO_F=0x01;
    /* Show "CAL" indicator */
    fill(BOX_X+10,BOX_Y+20,BOX_W-20,20,WHITE);
    dtext(BOX_X+10,BOX_Y+22,"CALIBRATING...",C_BLUE,WHITE,1);
    DIR_C();
    DELAYL(0xFF);
    DIS_T();
    EEPROM_SaveCalibration();
}

void VE_UP(void){
    uint8_t ucCount=0x65; uint8_t k;
    while(1){
        INCREMENT();
        if(*MM_IF==0x01){INCREMENT();INCREMENT();}
        VE_DISP();
        if(ucCount!=0x01) ucCount-=0x02;
        DLY_LOG(ucCount);
        k=ttp_read();
        if(!(k&K_UP)) break;
    }
}
void VE_DN(void){
    uint8_t ucCount=0x65; uint8_t k;
    while(1){
        DECREMENT();
        if(*MM_IF==0x01){DECREMENT();DECREMENT();}
        VE_DISP();
        if(ucCount!=0x01) ucCount-=0x02;
        DLY_LOG(ucCount);
        k=ttp_read();
        if(!(k&K_DN)) break;
    }
}
void VELO(void){
    uint16_t i; uint8_t k,pk=0;
    VE_DISP();
    while(1){
        for(i=0;i<0xFF;i++){
            DELAYL(0x20);
            k=ttp_read(); uint8_t np=k&~pk; pk=k;
            if(np&K_UP){VE_UP();break;}
            if(np&K_DN){VE_DN();break;}
        }
        if(i==0xFF) break;
    }
}

/* ================================================================
   MM/INCH TOGGLE — K3 replaces UP key from old IN_MM()
   ================================================================ */
void IN_MM(void){
    uint8_t k,pk=0;
    /* Show current mode */
    fill(BOX_X+10,BOX_Y+20,BOX_W-20,30,WHITE);
    dtext(BOX_X+10,BOX_Y+28,(*MM_IF==0x00)?"Mode: MM":"Mode: INCH",C_DARK,WHITE,2);
    /* Wait for K3 release */
    while(ttp_read()&K_MM);
    DELAY(0x20);
    while(1){
        k=ttp_read(); uint8_t np=k&~pk; pk=k;
        if(np&K_MM){
            *MM_IF=(*MM_IF==0x00)?0x01:0x00;
            fill(BOX_X+10,BOX_Y+20,BOX_W-20,30,WHITE);
            dtext(BOX_X+10,BOX_Y+28,(*MM_IF==0x00)?"Mode: MM":"Mode: INCH",C_DARK,WHITE,2);
            while(ttp_read()&K_MM);
            DELAY(0x20);
        }
        if(np&K_PROBE){ /* K5 = confirm */
            while(ttp_read()&K_PROBE);
            DELAY(0x20);
            break;
        }
    }
    if(*MM_IF==0x01){DP_1=0x00;DP_2=0x01;}
    else             {DP_1=0x01;DP_2=0x00;}
    EEPROM_SaveCalibration();
}

/* ================================================================
   TFT SPLASH SEQUENCE  (replaces SCROL)
   ================================================================ */
static void run_splash(void){
    /* Draw EDISON II bitmap while dark */
    win(0,0,SCR_W-1,SCR_H-1); CS_LO; DC_HI;
    for(uint16_t row=0;row<SCR_H;row++){
        uint16_t rb=row*SP_BPR;
        for(uint8_t b=0;b<SP_BPR;b++) sp_row[b]=pgm_read_byte(&sp_bmp[rb+b]);
        for(uint8_t b=0;b<SP_BPR;b++){
            uint8_t v=sp_row[b];
            spi_tx((v&0x80)?0:0xFF);spi_tx((v&0x80)?0:0xFF);
            spi_tx((v&0x40)?0:0xFF);spi_tx((v&0x40)?0:0xFF);
            spi_tx((v&0x20)?0:0xFF);spi_tx((v&0x20)?0:0xFF);
            spi_tx((v&0x10)?0:0xFF);spi_tx((v&0x10)?0:0xFF);
            spi_tx((v&0x08)?0:0xFF);spi_tx((v&0x08)?0:0xFF);
            spi_tx((v&0x04)?0:0xFF);spi_tx((v&0x04)?0:0xFF);
            spi_tx((v&0x02)?0:0xFF);spi_tx((v&0x02)?0:0xFF);
            spi_tx((v&0x01)?0:0xFF);spi_tx((v&0x01)?0:0xFF);
        }
    }
    CS_HI;
    for(uint8_t i=0;i<20;i++) _delay_ms(100);   /* hold 2 seconds */
    /* Fade out */
    lcd_cmd(0x53);lcd_dat(0x2C); lcd_cmd(0x51);lcd_dat(0xFF);
    for(int16_t b=255;b>=0;b-=8){lcd_cmd(0x51);lcd_dat((uint8_t)b);delay_ms(18);}
    lcd_cmd(0x51);lcd_dat(0x00);
    /* Draw instrument UI while dark */
    draw_instrument_ui();
    /* Fade in */
    for(uint16_t b=0;b<=255;b+=8){lcd_cmd(0x51);lcd_dat((uint8_t)b);delay_ms(18);}
    lcd_cmd(0x51);lcd_dat(0xFF);
    lcd_cmd(0x53);lcd_dat(0x00);  /* restore display pipeline */
}

/* ================================================================
   HARDWARE INIT
   ================================================================ */
static void clk24(void){
    CPU_CCP=CCP_IOREG_gc;
    CLKCTRL.OSCHFCTRLA=CLKCTRL_FRQSEL_24M_gc;
}
static void hw_init(void){
    /* Power latch — hold power on immediately */
    PORTA.DIRSET=P_10_PIN;
    PORTA.OUTSET=P_10_PIN;   /* latch ON — keep power */
    /* Power button input */
    PORTA.DIRCLR=SW_P_PIN;
    PORTA.PIN0CTRL=PORT_PULLUPEN_bm|PORT_ISC_FALLING_gc;
    /* TFT SPI control pins */
    PORTD.DIRSET=PIN0_bm|PIN1_bm|PIN2_bm|PIN3_bm|PIN4_bm|PIN5_bm;
    PORTD.OUTSET=PIN0_bm|PIN1_bm|PIN2_bm;   /* DC/RST/CS idle high */
    PORTD.OUTCLR=PIN3_bm|PIN4_bm|PIN5_bm;
    /* SPI MOSI=PA4, SCK=PA6, MISO=PA5(in), PA7(out) */
    PORTA.DIRSET=PIN4_bm|PIN6_bm|PIN7_bm;
    PORTA.DIRCLR=PIN5_bm;
    /* TTP229: SCL=PB1 output high idle, SDO=PB0 input */
    PORTB.DIRSET=PIN1_bm; PORTB.OUTSET=PIN1_bm;
    PORTB.DIRCLR=PIN0_bm; PORTB.PIN0CTRL=PORT_PULLUPEN_bm;
    /* Counter inputs — adjust per new PCB routing */
    PORTB.DIRCLR=PIN2_bm|PIN3_bm|PIN4_bm|PIN5_bm;
    PORTB.PIN2CTRL=PORT_PULLUPEN_bm; PORTB.PIN3CTRL=PORT_PULLUPEN_bm;
    PORTB.PIN4CTRL=PORT_PULLUPEN_bm; PORTB.PIN5CTRL=PORT_PULLUPEN_bm;
    PORTC.DIRCLR=PIN4_bm|PIN5_bm|PIN6_bm|PIN7_bm;
    PORTC.PIN4CTRL=PORT_PULLUPEN_bm; PORTC.PIN5CTRL=PORT_PULLUPEN_bm;
    PORTC.PIN6CTRL=PORT_PULLUPEN_bm; PORTC.PIN7CTRL=PORT_PULLUPEN_bm;
    PORTA.DIRCLR=PIN1_bm|PIN2_bm|PIN3_bm;
    PORTA.PIN1CTRL=PORT_PULLUPEN_bm;
    PORTA.PIN2CTRL=PORT_PULLUPEN_bm;
    PORTA.PIN3CTRL=PORT_PULLUPEN_bm;
    PORTF.DIRCLR=PIN1_bm; PORTF.PIN1CTRL=PORT_PULLUPEN_bm;
    /* Echo + ultrasonic outputs */
    PORTF.DIRSET=PIN0_bm;
    /* TCB0 timer */
    TCB0.CTRLB=TCB_CNTMODE_INT_gc;
    TCB0.CTRLA=TCB_CLKSEL_DIV1_gc;
    /* SPI */
    SPI0.CTRLB=SPI_SSD_bm;
    SPI0.CTRLA=SPI_MASTER_bm|SPI_ENABLE_bm|SPI_CLK2X_bm|SPI_PRESC_DIV4_gc;
}

/* ================================================================
   MAIN
   ================================================================ */
int main(void){
    clk24();
    hw_init();

    /* Init TFT first so display works immediately */
    lcd_init();

    /* Restore calibration state */
    if(*ucPOW==0xAA){
        if(SUM_CK()==0x00) INIT_M();
        else                INIT();
    } else {
        if(EEPROM_ValidateCalibration()==0x00){
            EEPROM_LoadCalibration(); *ucPOW=0xAA; INIT_M();
        } else {
            INIT();
        }
    }

    /* TFT splash → fade → instrument UI */
    run_splash();

    PORTA.INTFLAGS=INT_PIN; sei();   /* enable power button ISR */
    prev_keys=ttp_read();
    meas_tick=0; bat_tick=0; uptime=0;

    /* ── MAIN LOOP ──────────────────────────────────── */
    while(1){
        uint8_t keys=ttp_read();
        uint8_t np  =keys&~prev_keys;
        prev_keys=keys;

        for(uint8_t k=0;k<8;k++){
            if(keys&(1<<k)){if(key_hold[k]<255)key_hold[k]++;}
            else key_hold[k]=0;
            if(np&(1<<k)) key_cnt[k]++;
        }

        /* K8: toggle diagnostic mode from anywhere — instant, no blocking wait */
        if(np&K_DIAG){
            if(mode==0){
                /* Instrument → Diagnostic: straight to menu */
                mode=1;
                diag_go(SCR_DMENU);
                prev_keys=ttp_read();
            } else {
                /* Diagnostic → Instrument: disable brightness if active */
                if(diag_page==SCR_BRIGHT){ lcd_cmd(0x53); lcd_dat(0x00); }
                mode=0;
                draw_instrument_ui();
                DISPLAY();
                meas_tick=0;
                prev_keys=ttp_read();
            }
            continue;
        }

        /* ── MEASUREMENT MODE ────────────────────── */
        if(mode==0){
            /* 50ms base tick — keeps K8 responsive */
            _delay_ms(50);
            uptime++;

            /* Measurement runs every 5 ticks (~250ms) */
            if(++meas_tick>=5){
                meas_tick=0;
                Timer_Stop();
                THNESS();          /* ultrasonic measurement + DISPLAY() */
                Timer_Start(TIMER_VAL_SHORT);
                while(!Timer_Overflow());
                TIMER_OVERFLOW();
            }

            /* Battery display: decrement every 40 ticks (~2s) for demo */
            if(++bat_tick>=40){
                bat_tick=0;
                if(disp_batt>0) disp_batt--;
                else            disp_batt=100;
                draw_battery(disp_batt);
            }

            /* LO_BATT: auto power-off counter (keep original logic) */
            LO_BATT();

            /* Keys — checked every tick so K8 responds within 50ms */
            if(np&K_PROBE){
                Timer_Stop();
                PROBE();
                draw_instrument_ui();
                DISPLAY();
                continue;
            }
            if(np&K_CAL){
                Timer_Stop();
                DIST_C();
                draw_instrument_ui();
                DISPLAY();
                continue;
            }
            if(np&K_MM){
                IN_MM();
                draw_instrument_ui();
                DISPLAY();
            }

            /* Refresh uptime in topbar once per second */
            if(uptime%20==0) uptime_tick();
        }

        /* ── DIAGNOSTIC MODE ─────────────────────── */
        else {
            if(np&K_CAL){  /* K6 = back one level */
                if(diag_page==SCR_DMENU){
                    /* Back to instrument from menu */
                    mode=0;
                    draw_instrument_ui();
                    DISPLAY();
                    meas_tick=0;
                    prev_keys=ttp_read();
                } else {
                    /* Back to diag menu — disable brightness if leaving it */
                    if(diag_page==SCR_BRIGHT){ lcd_cmd(0x53); lcd_dat(0x00); }
                    diag_go(SCR_DMENU);
                }
                _delay_ms(20); continue;
            }
            switch(diag_page){
                case SCR_DMENU: {
                    uint8_t oc=menu_cur;
                    if(np&K_UP){if(menu_cur>0)menu_cur--;}
                    if(np&K_DN){if(menu_cur<4)menu_cur++;}
                    if(np&(K_UP|K_DN)) menu_move(oc);
                    if(np&K_PROBE){
                        if(MPAGE[menu_cur]==0xFF){
                            /* "Exit to Measurement" selected */
                            mode=0; draw_instrument_ui(); DISPLAY();
                            meas_tick=0; prev_keys=ttp_read(); continue;
                        }
                        diag_go(MPAGE[menu_cur]);
                    }
                    break;
                }
                case SCR_BRIGHT: {
                    static uint8_t bth=0; bth++;
                    uint8_t ch=0;
                    if((np&K_UP)||(bth%5==0&&(keys&K_UP))){if(bright_val<15){bright_val++;ch=1;}}
                    if((np&K_DN)||(bth%5==0&&(keys&K_DN))){if(bright_val>0) {bright_val--;ch=1;}}
                    if(ch){apply_bright(bright_val);draw_bright_ui();}
                    break;
                }
                case SCR_BTNTEST:
                    btntest_update(keys);
                    break;
                case SCR_VOLUME: {
                    uint8_t ch=0;
                    if(keys&K_MM) {if(vol_val>0)  {vol_val--;ch=1;}}
                    if(keys&K_RES){if(vol_val<100){vol_val++;ch=1;}}
                    if(ch) draw_vol_ui();
                    break;
                }
                case SCR_ABOUT:
                    break;
            }
            uptime++;
            if(uptime%33==0) uptime_tick();
            _delay_ms(30);
        }
    }
}
